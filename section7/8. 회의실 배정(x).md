# 문제설명
- 한개의 회의실이있는데 이를 사용하고자 하는 n개의 회의들에 대하여 회의실 사용표를 만들려고한다.
- 각 회의에 대해 시작시간과 끝나는시간이 주어지고, 각 회의가 겹치지않게 하면서 회의실를 사용할 수 있는 최대수의 회의를 찾아라.
- 단 회의를 한번 시작하면 중간에 중단될 수없으며 한 회의가 끝나는것과 동시에 다음회의가 시작될 수있다.


## 풀이
```html
  <script>
      function solution(meeting) {
        let answer = 1;
        meeting.sort((a, b) => {
          if (a[1] === b[1]) {
            return a[0] - b[0];
          } else {
            return a[1] - b[1];
          }
        });
        let lt = 0;
        for (let rt = 1; rt < meeting.length; rt++) {
          if (meeting[lt][1] <= meeting[rt][0]) {
            answer++;
            lt = rt;
          }
        }

        console.log(meeting);
        return answer;
      }

      let arr = [
        [3, 3],
        [1, 3],
        [2, 3],
      ];
      console.log(solution(arr));
    </script>
```


## 해설 
```html
    <script>
      function solution(meeting) {
        let answer = 0;
        meeting.sort((a, b) => {
          if (a[1] === b[1]) {
            return a[0] - b[0];
          } else {
            return a[1] - b[1];
          }
        });
        let et = 0; // et : endtime
        for (let x of meeting) {
          if (x[0] >= et) {
            answer++;
            et = x[1];
          }
        }

        return answer;
      }

      let arr = [
        [3, 3],
        [2, 3],
        [1, 3],
      ];
      console.log(solution(arr));
    </script>
```
