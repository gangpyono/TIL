# 아나그램(해쉬)
- 두 문자열의 나열 순서는 다르지만 그 구성이 일치하면 두단어를 아나그램이라고한다.
- AbaAeCe 와 baeeACA 는 아나그램이다.
- 두 단어가 아나그램이면 "YES"출력하고 그렇지않을경우 "NO"를 출력하는 프로그램을 작성해라.




## 풀이
```html
       <script>
            function solution(str1, str2){
                let answer="YES"; 
                let sH1 = new Map();
                let sH2 = new Map();
                
                for(let x of str1) {
                    if(sH1.has(x)) sH1.set(x,sH1.get(x)+1);
                    else sH1.set(x,1);
                }
                
                for(let x of str2) {
                    if(sH2.has(x)) sH2.set(x,sH2.get(x)+1);
                    else sH2.set(x,1);
                }

                for(x of str1) {
                    if(sH1.has(x) !== sH2.has(x)) {
                        answer = "No"; 
                        break;
                    }
                }
                
                console.log(sH1);
                console.log(sH2);
                return answer;
            }
         
            let a="AbaAeCe";
            let b="baeeACA";
            console.log(solution(a, b));
        </script>
```


## 해설 
```html
       <script>
            function solution(str1, str2){
                let answer="YES"; 
                let sH1 = new Map();
                
                
                for(let x of str1) {
                    if(sH1.has(x)) sH1.set(x,sH1.get(x)+1);
                    else sH1.set(x,1);
                }
                
                for(let x of str2) {
                    if(!sH1.has(x) || sH1.get(x)===0) return answer = "NO";
                    sH1.set(x,sH1.get(x)-1);
                }

                return answer;
            }
            // 1. 키가 존재하는지 확인
            // 2. 존재한다면 값을 1개 차감시킴.
            let a="abcCC";
            let b="Caaab";
            console.log(solution(a, b));
        </script>
```
